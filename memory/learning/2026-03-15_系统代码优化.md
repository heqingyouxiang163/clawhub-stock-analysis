## 3️⃣ 仓位管理优化

### 代码实现
```python
def calculate_position(emotion, pattern, profit_loss):
    # 基础仓位
    base = {"首板": 25, "连板": 40, "五日线": 35}.get(pattern, 20)
    
    # 情绪调整
    emotion_adj = {"冰点": 0.8, "启动": 1.0, "强势": 1.2, "高潮": 0.8}.get(emotion, 0.6)
    
    # 盈亏调整
    pl_adj = 1.1 if profit_loss > 10 else (0.8 if profit_loss < -5 else 1.0)
    
    position = min(max(base * emotion_adj * pl_adj, 0), 50)
    return round(position, 0)
```

def calculate_emotion_score(data):
    score = (
        data["limit_up"] * 0.25 +
        data["limit_down"] * 0.20 +
        data["consecutive"] * 0.15 +
        data["fail_rate"] * 0.15 +
        data["up_down_ratio"] * 0.10 +
        data["north_flow"] * 0.10 +
        data["super_large"] * 0.05
    )
    
    if score >= 80:
        return "高潮期"
    elif score >= 60:
        return "强势期"
    elif score >= 40:
        return "震荡期"
    elif score >= 20:
        return "启动期"
    else:
        return "冰点期"
