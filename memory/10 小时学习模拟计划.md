## 🔧 **技术实现**

### 雪球数据抓取脚本

```python
#!/usr/bin/env python3
# 雪球数据抓取脚本

import requests
import json
import time

class XueqiuDataFetcher:
    def __init__(self):
        self.session = requests.Session()
        self.base_url = "https://stock.xueqiu.com"
        
    def get_stock_data(self, symbol):
        """获取单只股票实时数据"""
        url = f"{self.base_url}/v5/stock/quote.json"
        params = {
            'symbol': symbol,
            'extend': 'detail'
        }
        
        try:
            response = self.session.get(url, params=params, timeout=5)
            data = response.json()
            
            # 数据校验
            if self.validate_data(data):
                return data['data']['quote']
            else:
                return None
                
        except Exception as e:
            print(f"抓取失败：{e}")
            return None
    
    def validate_data(self, data):
        """数据校验"""
        try:
            quote = data['data']['quote']
            
            # 涨跌幅校验
            change_pct = quote.get('percent', 0)
            if change_pct > 10 or change_pct < -10:
                print(f"涨跌幅异常：{change_pct}%")
                return False
            
            # 换手率校验
            turnover = quote.get('turnover_rate', 0)
            if turnover > 50 or turnover < 0:
                print(f"换手率异常：{turnover}%")
                return False
            
            # 价格校验
            current = quote.get('current', 0)
            high = quote.get('high', 0)
            low = quote.get('low', 0)
            
            if high < current:
                print(f"最高价异常：{high} < {current}")
                return False
            if low > current:
                print(f"最低价异常：{low} > {current}")
                return False
            
            return True
            
        except Exception as e:
            print(f"校验失败：{e}")
            return False
    
    def batch_fetch(self, symbols):
        """批量抓取"""
        results = {}
        for symbol in symbols:
            data = self.get_stock_data(symbol)
            if data:
                results[symbol] = data
            time.sleep(0.1)  # 避免频率限制
        return results
```

### 数据对比脚本

```python
#!/usr/bin/env python3
# 数据对比脚本

def compare_data_sources(xueqiu_data, mcp_data):
    """对比雪球和 MCP 数据"""
    differences = {}
    
    fields = ['current', 'percent', 'turnover_rate', 'volume', 'amount']
    
    for field in fields:
        xueqiu_val = xueqiu_data.get(field, 0)
        mcp_val = mcp_data.get(field, 0)
        
        if xueqiu_val != mcp_val:
            diff_pct = abs(xueqiu_val - mcp_val) / xueqiu_val * 100 if xueqiu_val > 0 else 0
            differences[field] = {
                'xueqiu': xueqiu_val,
                'mcp': mcp_val,
                'diff_pct': diff_pct
            }
    
    return differences
```
